Node.js — JWT Auth Middleware & Protected Routes
const jwt = require("jsonwebtoken");

// JWT Middleware
const protect = (req, res, next) => {
  const token = req.headers.authorization?.split(" ")[1];
  if (!token) return res.status(401).json({ msg: "Not authorized" });
  try {
    req.user = jwt.verify(token, process.env.JWT_SECRET);
    next();
  } catch { res.status(401).json({ msg: "Token expired" }); }
};

// Role-based middleware
const requireRole = (...roles) => (req, res, next) => {
  if (!roles.includes(req.user.role))
    return res.status(403).json({ msg: "Access denied" });
  next();
};

// Protected route — only managers
router.post("/tasks", protect, requireRole("manager"), async (req, res) => {
  const task = await Task.create({ ...req.body, createdBy: req.user._id });
  res.status(201).json(task);
});

// Reusable TaskCard Component
function TaskCard({ task, onComplete }) {
  const statusColors = {
    "In Progress": "#f59e0b",
    "Completed": "#22c55e",
    "Overdue": "#ef4444",
  };

  return (
    <div className="task-card">
      <h3>{task.title}</h3>
      <span style={{ color: statusColors[task.status] }}>
        {task.status}
      </span>
      <p>Due: {new Date(task.deadline).toLocaleDateString()}</p>
      <button onClick={() => onComplete(task._id)}>
        Mark Complete
      </button>
    </div>
  );
}

// Parent component with state
function TaskList() {
  const [tasks, setTasks] = useState([]);

  const handleComplete = (id) => {
    setTasks(prev => prev.map(t =>
      t._id === id ? { ...t, status: "Completed" } : t
    ));
  };

  return (
    <div>
      {tasks.map(task => (
        <TaskCard key={task._id} task={task} onComplete={handleComplete} />
      ))}
    </div>
  );
}

// useEffect for data fetching
function Dashboard() {
  const [tasks, setTasks] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const load = async () => {
      const res = await fetch("/api/tasks", {
        headers: { Authorization: `Bearer ${localStorage.getItem("token")}` }
      });
      const data = await res.json();
      setTasks(data);
      setLoading(false);
    };
    load();
  }, []); // runs once on mount

  if (loading) return <p>Loading...</p>;
  return <TaskList tasks={tasks} />;
}

// Protected Route — only allow certain roles
function ProtectedRoute({ role, children }) {
  const user = JSON.parse(localStorage.getItem("user"));
  if (!user) return <Navigate to="/login" />;
  if (user.role !== role) return <Navigate to="/unauthorized" />;
  return children;
}

// storage.js
const Storage = {
  get: () =>
    JSON.parse(localStorage
      .getItem("tasks")) || [],
  set: (tasks) =>
    localStorage.setItem(
      "tasks",
      JSON.stringify(tasks)
    )
};

// tasks.js — Core logic
let tasks = Storage.get();

const addTask = (text, priority) => {
  const task = {
    id: Date.now(),
    text,
    priority, // low|medium|high
    done: false,
    createdAt: new Date()
      .toISOString()
  };
  tasks.push(task);
  Storage.set(tasks);
  renderTasks();
};

const toggleTask = (id) => {
  tasks = tasks.map(t =>
    t.id === id
      ? { ...t, done: !t.done }
      : t
  );
  Storage.set(tasks);
  renderTasks();
};

const deleteTask = (id) => {
  tasks = tasks.filter(t =>
    t.id !== id
  );
  Storage.set(tasks);
  renderTasks();
};

const filterTasks = (type) => {
  if (type === "active")
    return tasks.filter(t => !t.done);
  if (type === "done")
    return tasks.filter(t => t.done);
  return tasks;
};
render.js — DOM & Events
// render.js — Dynamic DOM updates
const list = document
  .getElementById("task-list");
const counter = document
  .getElementById("task-count");

function renderTasks(filter="all") {
  const shown = filterTasks(filter);
  list.innerHTML = shown
    .map(t => `
    <li class="task
      ${t.done ? 'done' : ''}
      priority-${t.priority}">
      <input type="checkbox"
        ${t.done ? 'checked' : ''}
        onchange="toggleTask(${t.id})">
      <span>${t.text}</span>
      <span class="tag">
        ${t.priority}
      </span>
      <button onclick=
        "deleteTask(${t.id})">✕
      </button>
    </li>`).join("");

  counter.textContent =
    `${shown.filter(t=>!t.done).length}
    tasks remaining`;
}

// Form submission

// context/AuthContext.jsx
import { createContext, useContext,
  useState, useEffect } from "react";

const AuthCtx = createContext();

export function AuthProvider({ children }) {
  const [user, setUser] = useState(
    () => JSON.parse(
      localStorage.getItem("user"))
  );

  const login = (userData, token) => {
    localStorage.setItem("user",
      JSON.stringify(userData));
    localStorage.setItem("token", token);
    setUser(userData);
  };

  const logout = () => {
    localStorage.clear();
    setUser(null);
  };

  return (
    <AuthCtx.Provider
      value={{ user, login, logout }}>
      {children}
    </AuthCtx.Provider>
  );
}
export const useAuth = () =>
  useContext(AuthCtx);

// App.jsx — Role-based routing
import { Routes, Route, Navigate }
  from "react-router-dom";
import { useAuth } from
  "./context/AuthContext";

function ProtectedRoute({ role, el }) {
  const { user } = useAuth();
  if (!user)
    return <Navigate to="/login"/>;
  if (role && user.role !== role)
    return <Navigate to="/403"/>;
  return el;
}

export default function App() {
  return (
    <Routes>
      <Route path="/login"
        element=<Login/>/>
      <Route path="/employee/*"
        element=<ProtectedRoute
          role="employee"
          el=<EmployeeDashboard/>/>/>
      <Route path="/manager/*"
        element=<ProtectedRoute
          role="manager"
          el=<ManagerDashboard/>/>/>
      <Route path="/admin/*"
        element=<ProtectedRoute
          role="superadmin"
          el=<AdminDashboard/>/>/>
      <Route path="*"
        element=<Navigate
          to="/login"/>/>
    </Routes>
  );
}

📖 What is DevOps?

🔧 DevOps — Development + Operations

DevOps is a cultural and technical movement that combines software development (Dev) and IT operations (Ops) into a unified workflow. The goal is to shorten the development lifecycle, deliver high-quality software faster, and improve collaboration between teams that were traditionally siloed.

Before DevOps, developers wrote code and "threw it over the wall" to operations who deployed it. DevOps breaks this barrier — the same team is responsible for building, testing, deploying, and monitoring software.

"DevOps is not a tool, framework, or technology. It is a culture and mindset — one that requires developers and operations to work together throughout the entire software lifecycle."
🔄 Continuous Integration (CI)
Developers frequently merge code changes into a shared repository. Each merge triggers automated builds and tests to catch bugs early — before they reach production. CI prevents "integration hell" caused by long-lived branches.
🚀 Continuous Delivery (CD)
After CI passes, the code is automatically deployed to staging or production environments. Every commit can potentially be released to users — enabling fast, reliable, low-risk deployments. TaskBridge uses Vercel for automatic CD on every push to main.
📦 Infrastructure as Code (IaC)
Managing infrastructure (servers, databases, networks) using configuration files instead of manual processes. Tools like Terraform, Docker, and Kubernetes allow teams to version-control their infrastructure just like application code.
📊 Monitoring & Observability
After deployment, the system must be continuously monitored for errors, performance issues, and downtime. Tools like Grafana, Prometheus, and Datadog help teams detect issues before users do and respond quickly.

⚙ How We Applied DevOps in TaskBridge

🐙

Git & GitHub — Version Control

Every line of code is tracked with Git. We used feature branches for each task, pull requests for code review, and protected the main branch from direct pushes. The full history of every change is preserved in our GitHub repositories.

Git — Branching Workflow Used in TaskBridge
# 1. Start a new feature
git checkout -b feature/task-assignment
# make changes...
git add .
git commit -m "feat: add task assignment with deadline"
git push origin feature/task-assignment

# 2. Create Pull Request on GitHub → code review → merge

# 3. Daily sync with main branch
git checkout main
git pull origin main           # get latest changes from teammates
git checkout feature/my-branch
git merge main                 # resolve any conflicts early

# 4. Useful inspection commands
git log --oneline --graph      # visual branch history
git diff main..feature/login   # what changed vs main?
git blame src/components/Nav.jsx  # who wrote this line?
git stash                      # save WIP before switching branches
🚀

Vercel — Continuous Deployment (CD)

Vercel watches our GitHub repository. Every push to the main branch automatically builds and deploys the React app — zero manual steps. Preview deployments are created for every pull request so we can test before merging. Live at: taskbridge-frontend-sandy.vercel.app

🍃

MongoDB Atlas — Cloud Database Operations

MongoDB Atlas handles database DevOps for us — automated backups, scaling, monitoring, and security patches. We configured IP access lists, database users with least-privilege roles, and connection string secrets stored in environment variables — never in code.

🔐

Environment Variables — Secrets Management

No secrets are ever committed to Git. MongoDB URI, JWT secret, Google OAuth keys, and email credentials are stored in .env files locally and in Vercel's encrypted environment panel for production.
